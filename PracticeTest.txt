             }

   {
         Double malecount=0.0;
         Double femalecount=0.0;
         long male=0;
         long female=0;
         for(Candidate c:list)
         {
            if(c.getGender().equals("Male"))
             {
               male+=c.getAge();
               malecount++;
             }
             else
             {
               female+=c.getAge();
               femalecount++;
             }
         }
         HashMap<String,Double>hm=new HashMap<>();
         if(female!=0)
          hm.put("Female",(Double)(female/femalecount));
        if(male!=0)
          hm.put("Male",(Double)(male/malecount));
        return hm;

   }

     public static Map<String,Long> countCandidatesDepartmentWise(List<Candidate> list)
     {
          HashMap<String,Long>hm=new HashMap<>();
          Long l=1L;
          for(Candidate c:list)
          {
               String curr_dept=c.getDepartment();
               if(hm.containsKey(curr_dept))
               {
                   hm.put(curr_dept,hm.get(curr_dept)+1);
               }
               else
               {
                   hm.put(curr_dept,l);
               }
          }
          return hm;
     }
